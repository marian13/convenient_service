def b(*args)
  require "byebug"
  require "byebug/core"

  ::Byebug.attach

  args.one? ? args.first : args
end

def bb(*args)
  require "debug"

  binding.break(up_level: 2)

  args.one? ? args.first : args
end

def ds(first_string, second_string)
  require "diffy"

  puts ::Diffy::Diff.new(first_string, second_string).to_s(:color)
end
