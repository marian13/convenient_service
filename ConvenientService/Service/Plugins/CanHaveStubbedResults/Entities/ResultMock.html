def register(&block)
  Commands::SetServiceStubbedResult[service: service_class, arguments: arguments, result: result]

  return self unless block

  begin
    yield
  ensure
    Commands::DeleteServiceStubbedResult[service: service_class, arguments: arguments]
  end
end
