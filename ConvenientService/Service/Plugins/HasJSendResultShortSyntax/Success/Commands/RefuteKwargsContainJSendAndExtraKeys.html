def call
  return if [:data, :message, :code].none? { |key| kwargs.has_key?(key) }

  return if kwargs.keys.difference([:data, :message, :code]).none?

  ::ConvenientService.raise Exceptions::KwargsContainJSendAndExtraKeys.new
end
