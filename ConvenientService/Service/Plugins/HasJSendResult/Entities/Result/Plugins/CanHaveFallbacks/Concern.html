def with_error_fallback(raise_when_missing: true)
  return self unless status.unsafe_error?

  return service.__send__(:fallback_error_result) if has_own_instance_method?(service.class, :fallback_error_result, private: true)
  return service.__send__(:fallback_result) if has_own_instance_method?(service.class, :fallback_result, private: true)

  ::new(result: self, status: :error) if raise_when_missing
end

def with_failure_and_error_fallback(...)
  case status.to_sym
  when :success
    self
  when :failure
    with_failure_fallback(...)
  else # :error
    with_error_fallback(...)
  end
end

def with_fallback_for(statuses_to_fallback, **kwargs)
  case status.to_sym
  when :success
    self
  when :failure
    wrap(statuses_to_fallback).include?(:failure) ? with_failure_fallback(**kwargs) : self
  else # error
    wrap(statuses_to_fallback).include?(:error) ? with_error_fallback(**kwargs) : self
  end
end
