def call
  webrick_request = ::WEBrick::HTTPRequest.new(::WEBrick::Config::HTTP)

  webrick_request.parse(::StringIO.new(http_string))

  webrick_request.body
rescue ::WEBrick::HTTPStatus::BadRequest
  nil
end
