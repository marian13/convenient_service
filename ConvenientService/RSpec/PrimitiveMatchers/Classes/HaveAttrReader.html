def matches?(object)
  @object = object

  Utils.with_one_time_object do |one_time_object|
    copy = object.dup

    copy.instance_variable_set(instance_variable_name, one_time_object)

    Utils.safe_send(copy, attr_name) == one_time_object
  end
end
