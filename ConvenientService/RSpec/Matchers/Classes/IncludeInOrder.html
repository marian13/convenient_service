def matches?(string)
  @string = string

  return false if keywords.empty?

  scanner = ::StringScanner.new(string)

  keywords
    .map { |keyword| keyword.instance_of?(::String) ? ::Regexp.new(::Regexp.escape(keyword)) : keyword }
    .all? { |keyword| scanner.scan_until(keyword) }
end
