def update_expected_return_value_block(*args, &block)
  values.delete(:custom_return_value)

  values[:expected_return_value_block] =
    if args.any? && block
      ::ConvenientService.raise Exceptions::ReturnCustomValueChainingInvalidArguments.new
    elsif args.any?
      proc { args.first }
    elsif block
      block
    else
      ::ConvenientService.raise Exceptions::ReturnCustomValueChainingInvalidArguments.new
    end
end
