def call
  return "" if arguments.null_arguments?

  text =
    if args.any? && kwargs.any?
      "(#{printable_args}, #{printable_kwargs})"
    elsif args.any?
      "(#{printable_args})"
    elsif kwargs.any?
      "(#{printable_kwargs})"
    else
      "()"
    end

  text += " #{printable_block}" if block

  text
end
