def matches?(object)
  @object = object

  ##
  # IMPORTANT: A copy is created in order to be thread safe.
  #
  copy = object.dup

  copy.public_send("#{attr_name}=", :custom_value)

  copy.instance_variable_get("@#{attr_name}") == :custom_value
end
