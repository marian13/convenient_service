def call(env)
  return env if stack.empty?

  app =
    stack
      .map { |middleware| proc { |app| middleware.new(app) } }
      .reverse
      .reduce(Custom::Constants::INITIAL_MIDDLEWARE) { |a, e| e[a] }

  app.call(env)
end
