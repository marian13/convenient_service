def [](key)
  case key
  when Integer then args[key]
  when Symbol then kwargs[key]
  else ::ConvenientService.raise Exceptions::InvalidKeyType.new(key: key)
  end
end

def deconstruct_keys(keys)
  keys ||= [:args, :kwargs, :block]

  keys.each_with_object({}) do |key, hash|
    case key
    when :args
      hash[key] = args
    when :kwargs
      hash[key] = kwargs
    when :block
      hash[key] = block
    end
  end
end
