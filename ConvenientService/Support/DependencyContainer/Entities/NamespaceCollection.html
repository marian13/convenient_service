def find_by(name: Support::NOT_PASSED)
  rules = []

  rules << ->(namespace) { namespace.name.to_s == name.to_s } if name != Support::NOT_PASSED

  condition = Utils::Proc.conjunct(rules)

  namespaces.find(&condition)
end
